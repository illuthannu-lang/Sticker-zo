<script>
const canvas = document.getElementById("canvas");
const sizeSlider = document.getElementById("sizeSlider");

function addSticker(emoji) {
    const sticker = document.createElement("div");
    sticker.className = "sticker spawn";
    sticker.textContent = emoji;

    const size = sizeSlider.value;
    sticker.style.fontSize = size + "px";

    sticker.style.left = Math.random() * 400 + "px";
    sticker.style.top = Math.random() * 300 + "px";
    sticker.style.transform = "rotate(" + (Math.random()*30-15) + "deg)";

    makeDraggable(sticker);
    canvas.appendChild(sticker);

    saveZoo();
}

function clearCanvas() {
    canvas.innerHTML = "";
    localStorage.removeItem("zoo");
}

function changeBackground(val) {
    const colors = val.split(",");
    document.body.style.background = `linear-gradient(0deg, ${colors[0]}, ${colors[1]})`;
}

function makeDraggable(el) {
    let offsetX, offsetY, isDown = false;

    el.addEventListener("pointerdown", (e) => {
        isDown = true;
        offsetX = e.offsetX;
        offsetY = e.offsetY;
        el.setPointerCapture(e.pointerId);
    });

    el.addEventListener("pointermove", (e) => {
        if (!isDown) return;
        const rect = canvas.getBoundingClientRect();
        el.style.left = (e.clientX - rect.left - offsetX) + "px";
        el.style.top = (e.clientY - rect.top - offsetY) + "px";
        saveZoo();
    });

    el.addEventListener("pointerup", () => isDown = false);
}

function saveZoo() {
    const stickers = [];
    document.querySelectorAll(".sticker").forEach(s => {
        stickers.push({
            emoji: s.textContent,
            left: s.style.left,
            top: s.style.top,
            size: s.style.fontSize,
            transform: s.style.transform
        });
    });
    localStorage.setItem("zoo", JSON.stringify(stickers));
}

function loadZoo() {
    const saved = JSON.parse(localStorage.getItem("zoo") || "[]");
    saved.forEach(s => {
        const sticker = document.createElement("div");
        sticker.className = "sticker";
        sticker.textContent = s.emoji;
        sticker.style.left = s.left;
        sticker.style.top = s.top;
        sticker.style.fontSize = s.size;
        sticker.style.transform = s.transform;
        makeDraggable(sticker);
        canvas.appendChild(sticker);
    });
}

loadZoo();
</script>
